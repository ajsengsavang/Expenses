Connect Google Sheets
This links your expense tracker to a Google Sheet so both you and Manda can access it from any device. Takes about 5 minutes to set up.
1
Create the Google Sheet
Go to sheets.google.com, create a new blank sheet. Name it "Sengsavang Expenses" or anything you like. Copy the Sheet ID from the URL — it's the long string between /d/ and /edit.
2
Create a Google Apps Script
In the sheet, go to Extensions → Apps Script. Delete any existing code and paste the code below, then click Deploy → New Deployment → Web App. Set "Who has access" to Anyone. Copy the Web App URL.
// Paste this in Google Apps Script const SHEET_ID = 'YOUR_SHEET_ID_HERE'; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(SHEET_ID); let sheet = ss.getSheetByName(data.tab); if (!sheet) { sheet = ss.insertSheet(data.tab); sheet.appendRow(['ID','Date','Vendor/Restaurant','Category/Type','Amount','Extra1','Extra2','Notes','PaidBy','Timestamp']); } if (data.action === 'add') { sheet.appendRow([ data.row.id, data.row.date, data.row.vendor, data.row.category, data.row.amount, data.row.extra1||'', data.row.extra2||'', data.row.notes||'', data.row.paidBy||'', new Date().toISOString() ]); } else if (data.action === 'delete') { const rows = sheet.getDataRange().getValues(); for (let i = rows.length - 1; i >= 1; i--) { if (String(rows[i][0]) === String(data.id)) { sheet.deleteRow(i+1); break; } } } return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { const tab = e.parameter.tab; const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName(tab); if (!sheet) return ContentService.createTextOutput(JSON.stringify({ok:true,rows:[]})).setMimeType(ContentService.MimeType.JSON); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const data = rows.slice(1).map(r => Object.fromEntries(headers.map((h,i) => [h,r[i]]))); return ContentService.createTextOutput(JSON.stringify({ok:true,rows:data})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON); } }
